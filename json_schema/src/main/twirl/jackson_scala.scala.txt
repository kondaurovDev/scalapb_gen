@(schemas: List[json_schema.Models.JsonSchema])
@import org.json4s.jackson._

package json_schema

import com.fasterxml.jackson.databind.{JsonNode, ObjectMapper}

case class JsonSchema(
  id: String,
  schema: JsonNode
)

object JsonSchema {

	val mapper = new ObjectMapper();

	def parseJson(str: String): JsonNode = {
		mapper.readTree(str)
	}

@for(schema <- schemas) {
	val @{schema.id.replace('.', '_')} = JsonSchema("@schema.id", parseJson(""" @compactJson(schema.schema) """));
}

}
